var rules = [{
  url: /.*radu.*/,
  name: "Rule using a Lib function",
  fields: [{
    selector: "#form-o-fill-prompt-fill input",
    value: function(e, data) {
      var input = prompt("Paste data here:");
      input.trim().split(/\n/g).forEach(function(row, rowIndex) {
        row.trim().split(/\s/).forEach(function(cell, cellIndex) {
          var $fieldToFill = $("input[name='val[" + rowIndex+ "][" + cellIndex + "]']");
          $fieldToFill.val(cell);
        });
      });
      
      return null;
    }
  }]
}];
